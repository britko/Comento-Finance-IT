create table stock_invest_caution
(
    code_name text       null COMMENT '종목코드',
    code      varchar(6) null COMMENT '종목명',
    type      text       null COMMENT '공시일',
    post_date date       null COMMENT '지정일',
    fix_date  date       null COMMENT '해제일'
) COMMENT '투자 주의 종목';

create index ix_stock_invest_caution_index
    on stock_invest_caution (`code`);
